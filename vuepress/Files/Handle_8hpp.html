namespace Engine {
    namespace Graphics {
        class Handle {
          public:
            Handle() = default;
            virtual ~Handle() = default;

            int getHandle() const {
                return handle;
            }

          protected:
            // OpenGL handle
            int handle;
        };

        inline int getHandle(Handle& handle) {
            return handle.getHandle();
        }
